  <script>
    let state = { lead: {}, audit: {}, results: [] };

    // Stripe Checkout + Netlify Forms
    function startCheckout() {
      const name = document.getElementById('name').value;
      const email = document.getElementById('email').value;
      if(!name || !email) { alert("Please enter your name and email."); return; }

      state.lead = { name, email };

      // Save lead to Netlify Forms
      const formData = new FormData();
      formData.append("form-name", "lead-capture");
      formData.append("name", name);
      formData.append("email", email);

      fetch("/", { method: "POST", body: formData }).catch(() => {});

      // Call Netlify Stripe function
      fetch("/.netlify/functions/createCheckout", {
        method: "POST",
        headers: { "Content-Type": "application/json" },
        body: JSON.stringify({ name, email })
      })
      .then(res => res.json())
      .then(data => {
        if(data.url) {
          window.location.href = data.url; // redirect to Stripe Checkout
        } else {
          alert("Checkout error");
        }
      })
      .catch(() => alert("Checkout request failed"));
    }

    // Run audit logic
    function runAudit() {
      const bizType = document.getElementById('bizType').value;
      const challenges = document.getElementById('challenges').value.toLowerCase();

      const gaps = [];
      const recos = [];

      if(challenges.includes("proposals")) {
        gaps.push("You need help with proposals & decks.");
        recos.push("Gamma - create professional proposals fast.");
      }
      if(challenges.includes("research")) {
        gaps.push("You need better fact-checking & data.");
        recos.push("Perplexity - get real statistics with citations.");
      }
      if(challenges.includes("docs")) {
        gaps.push("You need structured long-form documents.");
        recos.push("Claude - draft business docs & strategies.");
      }
      if(challenges.includes("automation") || challenges.includes("reporting")) {
        recos.push("Zapier - bonus automation tool to save hours.");
      }

      state.audit = { bizType, challenges };
      state.results = recos;

      document.getElementById("audit").style.display = "none";
      document.getElementById("results").style.display = "block";

      document.getElementById("gapSummary").innerHTML = gaps.map(g => `<p>${g}</p>`).join("");
      document.getElementById("recoList").innerHTML = recos.map(r => `<div class="tool-card"><h3>${r}</h3></div>`).join("");
    }

    // Call Netlify PDF function
    function downloadPDF() {
      fetch("/.netlify/functions/generateReport", {
        method: "POST",
        headers: { "Content-Type": "application/json" },
        body: JSON.stringify({
          lead: state.lead,
          audit: state.audit,
          results: state.results
        })
      })
      .then(res => res.blob())
      .then(blob => {
        const url = window.URL.createObjectURL(blob);
        const a = document.createElement("a");
        a.href = url;
        a.download = "AI-Audit-Report.pdf";
        a.click();
      })
      .catch(() => alert("PDF generation failed"));
    }
  </script>
